1 ]=> (define (analyze array)
  (do ((i 0)
       (symbols '()))
      ((eqv? i (length array)) symbols)
    (cond ((symbol? (list-ref array i))
	   (set! symbols (append symbols '(ATOM))))
	  (else (set! symbols (append symbols '(LIST)))))
    (set! i (+ i 1)))
)
;Value: analyze
